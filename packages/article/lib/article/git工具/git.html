git 忽略追踪某个文件 #

bash
# 忽略追踪 ./project.config.json
git update-index --assume-unchanged ./project.config.json

# 取消忽略
git update-index --no-assume-unchanged ./project.config.json

git commit 规范 #

feat: 新功能

fix: 修复 bug

docs: 文档

style: 格式

refactor: 重构

perf: 优化，性能、体验

test: 测试

revert: 回滚

merge: 代码合并

chore: 构建

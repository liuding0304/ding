# 冒泡排序

``` js
var arr= [12,234,34,46,34,2,45,83,639];

var temp =0;

for(var i = 0; i < arr.length-1; ++i){ // 遍历每一个元素
    for(var n = i+1; n < arr.length; ++n){  // 每一个元素和后面所有的比较
        if(arr[i]<arr[n]){
            temp = arr[n];
            arr[n] = arr[i];
            arr[i] = temp;
        }
    }
}
console.log(arr);

var arr2 = [12,234,34,46,34,2,45,83,9];

for(var i = 0; i < arr2.length-1; ++i){  // 遍历每一个元素
    for(var n = 1; n < arr2.length -i; ++n){  // 每一个元素和后面所有的比较
        if(arr2[n]<arr2[n-1]){
            temp = arr2[n];
            arr2[n] = arr2[n-1];
            arr2[n-1] = temp;
        }
    }
}
console.log(arr2);
```
